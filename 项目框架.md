# NLP 情感分析 — 课程项目框架

> 🎯 **目标**：一天（~8h）完成一个有训练、有对比、有界面的完整 AI 情感分析项目  
> 💻 **环境**：Python 3.8+，建议有 NVIDIA GPU（4GB 显存就够），CPU 也能跑只是慢  
> 📦 **核心技术**：Transformers 微调 + scikit-learn 基线 + Streamlit 界面

---

## 1. 项目概述

### 做什么
一个 **中文文本情感三分类** 系统：输入一段中文文本，判断情感为 **正面 / 中性 / 负面**。

### 最终产出
```
1. 一个训练好的 BERT 微调模型
2. 一个 Web 界面（Streamlit），输入文本 → 显示情感 + 置信度
3. 一份对比实验结论（传统 ML vs BERT 微调）
4. 项目源码（5-6 个文件）
```

### 为什么这样做有价值
- 不是单纯调包，而是 **自己微调了模型**（课程项目核心得分点）
- 有 **对比实验**：传统机器学习 vs 深度学习，能写出有说服力的报告
- 有 **可视化界面**，演示效果好

### 步骤 2：传统 ML 基线 `02_baseline_ml.py`（1 h）

**目标**：用 TF-IDF + 逻辑回归做一个基线模型，作为后续 BERT 的对比参照。

```python
"""
步骤2：传统机器学习基线
- TF-IDF 特征提取
- 逻辑回归 / SVM / 朴素贝叶斯 对比
- 保存最优模型
"""
from datasets import load_dataset
from sklearn.feature_extraction.text import TfidfVectorizer
from sklearn.linear_model import LogisticRegression
from sklearn.svm import LinearSVC
from sklearn.naive_bayes import MultinomialNB
from sklearn.metrics import accuracy_score, classification_report
from sklearn.pipeline import Pipeline
import pickle
import jieba  # 中文分词
import os

# 加载数据
dataset = load_dataset("seawavetech/chnsenticorp")
X_train = dataset['train']['text']
y_train = dataset['train']['label']
X_test = dataset['test']['text']
y_test = dataset['test']['label']

print(f"训练集: {len(X_train)} 条, 测试集: {len(X_test)} 条")

# 中文需要分词，用 jieba
def tokenize(text):
    return ' '.join(jieba.cut(text))

X_train_cut = [tokenize(t) for t in X_train]
X_test_cut = [tokenize(t) for t in X_test]

# 三个模型对比
models = {
    'LogisticRegression': LogisticRegression(max_iter=1000),
    'LinearSVM': LinearSVC(max_iter=1000),
    'NaiveBayes': MultinomialNB(),
}

best_score = 0
best_model = None

for name, model in models.items():
    pipe = Pipeline([
        ('tfidf', TfidfVectorizer(ngram_range=(1, 2), max_features=10000)),
        ('clf', model),
    ])
    pipe.fit(X_train_cut, y_train)
    
    y_pred = pipe.predict(X_test_cut)
    acc = accuracy_score(y_test, y_pred)
    
    print(f"\n{'='*40}")
    print(f"模型: {name}")
    print(f"准确率: {acc:.4f}")
    print(classification_report(y_test, y_pred, target_names=['负面', '正面']))
    
    if acc > best_score:
        best_score = acc
        best_model = (name, pipe)

# 保存最优模型
os.makedirs('models', exist_ok=True)
with open('models/tfidf_vectorizer.pkl', 'wb') as f:
    pickle.dump(best_model[1].named_steps['tfidf'], f)
with open('models/ml_model.pkl', 'wb') as f:
    pickle.dump(best_model[1].named_steps['clf'], f)

print(f"\n✅ 最优模型: {best_model[0]}, 准确率: {best_score:.4f}")
print("模型已保存至 models/ 目录")
```

> **需要额外安装**：`pip install jieba`
> 
> **预期准确率**：约 85%-88%（这个数据集比较简单）
> 
> 这个准确率就是 BERT 需要超越的基线。

### 步骤 3：BERT 微调训练 `03_train_bert.py`（2.5 h）

**目标**：用 `bert-base-chinese` 微调，这是项目的核心。

```python
"""
步骤3：BERT 微调训练
- 加载 bert-base-chinese 预训练模型
- 在 chnsenticorp 上微调
- 保存最优 checkpoint
"""
from datasets import load_dataset
from transformers import (
    AutoTokenizer,
    AutoModelForSequenceClassification,
    TrainingArguments,
    Trainer,
    EarlyStoppingCallback,
)
import numpy as np
from sklearn.metrics import accuracy_score, f1_score
import os

# ── 1. 加载数据 ──────────────────────────────────
dataset = load_dataset("seawavetech/chnsenticorp")
num_labels = 2  # 二分类

print(f"训练: {len(dataset['train'])}, 测试: {len(dataset['test'])}")

# ── 2. 加载 Tokenizer ───────────────────────────
model_name = "bert-base-chinese"
tokenizer = AutoTokenizer.from_pretrained(model_name)

def preprocess(examples):
    """分词处理"""
    return tokenizer(
        examples['text'],
        truncation=True,
        padding='max_length',
        max_length=128,        # 中文短文本 128 足够
    )

encoded = dataset.map(preprocess, batched=True)

# ── 3. 加载预训练模型 ───────────────────────────
model = AutoModelForSequenceClassification.from_pretrained(
    model_name,
    num_labels=num_labels,
)

# ── 4. 定义评估指标 ─────────────────────────────
def compute_metrics(eval_pred):
    logits, labels = eval_pred
    preds = np.argmax(logits, axis=-1)
    acc = accuracy_score(labels, preds)
    f1 = f1_score(labels, preds, average='macro')
    return {'accuracy': acc, 'f1': f1}

# ── 5. 训练参数 ─────────────────────────────────
training_args = TrainingArguments(
    output_dir='./outputs',
    evaluation_strategy='steps',
    eval_steps=100,
    save_steps=500,
    logging_steps=50,
    num_train_epochs=5,
    per_device_train_batch_size=16,    # 显存不够改 8
    per_device_eval_batch_size=32,
    learning_rate=2e-5,
    weight_decay=0.01,
    warmup_ratio=0.1,
    load_best_model_at_end=True,
    metric_for_best_model='accuracy',
    save_total_limit=2,
    fp16=False,                       # 有 GPU 改成 True 加速
    report_to='none',
)

# ── 6. 训练 ─────────────────────────────────────
trainer = Trainer(
    model=model,
    args=training_args,
    train_dataset=encoded['train'],
    eval_dataset=encoded['test'],
    tokenizer=tokenizer,
    compute_metrics=compute_metrics,
    callbacks=[EarlyStoppingCallback(early_stopping_patience=2)],
)

print("\n🚀 开始训练...")
trainer.train()

# ── 7. 评估 ─────────────────────────────────────
results = trainer.evaluate()
print(f"\n📊 最终测试集结果:")
for k, v in results.items():
    print(f"  {k}: {v:.4f}")

# ── 8. 保存模型 ─────────────────────────────────
save_path = './models/bert-finetuned'
os.makedirs(save_path, exist_ok=True)
model.save_pretrained(save_path)
tokenizer.save_pretrained(save_path)
print(f"\n✅ 模型已保存至: {save_path}")
```

> **运行**：`python 03_train_bert.py`
> 
> **时间**：CPU 可能需要 1-2 小时，GPU（T4）约 10-15 分钟
> 
> **预期准确率**：约 92%-95%，明显高于 ML 基线
> 
> **显存不够怎么办**：把 `per_device_train_batch_size` 改成 8，`max_length` 改成 64。

### 步骤 4：评估对比 `04_evaluate.py`（30 min）

**目标**：把 ML 基线和 BERT 放到一起对比，画混淆矩阵，评估结果一目了然。

```python
"""
步骤4：模型评估 & 对比
- 加载 ML 基线和 BERT 模型
- 在测试集上对比
- 画混淆矩阵
"""
from datasets import load_dataset
from transformers import AutoTokenizer, AutoModelForSequenceClassification
from transformers import pipeline
from sklearn.metrics import (
    accuracy_score, f1_score, classification_report, confusion_matrix
)
import pickle
import jieba
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns
import pandas as pd
import time

# ── 1. 加载测试集 ───────────────────────────────
dataset = load_dataset("seawavetech/chnsenticorp")
X_test = dataset['test']['text']
y_test = dataset['test']['label']

# ── 2. 评估 ML 模型 ─────────────────────────────
with open('models/ml_model.pkl', 'rb') as f:
    ml_model = pickle.load(f)
with open('models/tfidf_vectorizer.pkl', 'rb') as f:
    tfidf = pickle.load(f)

def tokenize(text):
    return ' '.join(jieba.cut(text))

X_test_cut = [tokenize(t) for t in X_test]
X_test_tfidf = tfidf.transform(X_test_cut)

t0 = time.time()
y_ml_pred = ml_model.predict(X_test_tfidf)
ml_time = (time.time() - t0) / len(X_test) * 1000

ml_acc = accuracy_score(y_test, y_ml_pred)
ml_f1 = f1_score(y_test, y_ml_pred, average='macro')

# ── 3. 评估 BERT 模型 ───────────────────────────
model_path = './models/bert-finetuned'
tokenizer = AutoTokenizer.from_pretrained(model_path)
model = AutoModelForSequenceClassification.from_pretrained(model_path)
pipe = pipeline("text-classification", model=model, tokenizer=tokenizer, device=-1)

t0 = time.time()
bert_results = pipe(list(X_test), batch_size=32)
bert_time = (time.time() - t0) / len(X_test) * 1000

y_bert_pred = []
for r in bert_results:
    # 输出格式因模型而异，需适配
    if 'LABEL_1' in r['label'] or r['label'] == '1':
        y_bert_pred.append(1)
    else:
        y_bert_pred.append(0)

bert_acc = accuracy_score(y_test, y_bert_pred)
bert_f1 = f1_score(y_test, y_bert_pred, average='macro')

# ── 4. 打印对比 ─────────────────────────────────
print("\n" + "=" * 60)
print("                    模型对比结果")
print("=" * 60)
print(f"{'指标':<20} {'传统ML':<15} {'BERT微调':<15}")
print("-" * 60)
print(f"{'准确率':<20} {ml_acc:<15.4f} {bert_acc:<15.4f}")
print(f"{'Macro F1':<20} {ml_f1:<15.4f} {bert_f1:<15.4f}")
print(f"{'单条推理耗时':<20} {ml_time:<14.1f}ms {bert_time:<14.1f}ms")
print("=" * 60)

# ── 5. 画混淆矩阵 ───────────────────────────────
fig, axes = plt.subplots(1, 2, figsize=(12, 5))

for ax, y_pred, title in zip(
    axes,
    [y_ml_pred, y_bert_pred],
    ['传统 ML', 'BERT 微调']
):
    cm = confusion_matrix(y_test, y_pred)
    sns.heatmap(cm, annot=True, fmt='d', cmap='Blues', ax=ax,
                xticklabels=['负面', '正面'],
                yticklabels=['负面', '正面'])
    ax.set_title(f'{title}\n准确率: {accuracy_score(y_test, y_pred):.2%}')
    ax.set_xlabel('预测')
    ax.set_ylabel('真实')

plt.tight_layout()
plt.savefig('models/model_comparison.png', dpi=150)
print("\n📊 混淆矩阵已保存至 models/model_comparison.png")

# ── 6. 找出 BERT 预测错的样本 ───────────────────
errors = []
for i, (true, pred) in enumerate(zip(y_test, y_bert_pred)):
    if true != pred:
        errors.append({
            'text': X_test[i],
            '真实': '正面' if true == 1 else '负面',
            '预测': '正面' if pred == 1 else '负面',
        })

error_df = pd.DataFrame(errors)
error_df.to_csv('models/bert_errors.csv', index=False)
print(f"\n❌ BERT 预测错误: {len(errors)} 条 (错误率: {len(errors)/len(y_test):.2%})")
print("错误样本已保存至 models/bert_errors.csv，可用于报告分析")
```

> **运行**：`python 04_evaluate.py`
> 
> **预期**：BERT 准确率比 ML 基线高 5-8 个百分点，混淆矩阵一目了然。

### 步骤 5：Streamlit 界面 `app.py`（1.5 h）

**目标**：做一个好看的 Web 界面，能实际输入文本看分析结果。

```python
"""
情感分析 Web 界面
运行: streamlit run app.py
"""
import streamlit as st
from transformers import AutoTokenizer, AutoModelForSequenceClassification
from transformers import pipeline
import torch
import pickle
import jieba

# ── 页面配置 ────────────────────────────────────
st.set_page_config(
    page_title="中文情感分析",
    page_icon="🔍",
    layout="centered",
)

st.title("🔍 中文情感分析系统")
st.caption("基于 BERT 微调 | ChnSentiCorp 数据集 | 课程项目")

# ── 加载模型（缓存，只加载一次）─────────────────
@st.cache_resource
def load_models():
    """加载 BERT 和 ML 模型"""
    # BERT
    tokenizer = AutoTokenizer.from_pretrained('./models/bert-finetuned')
    model = AutoModelForSequenceClassification.from_pretrained('./models/bert-finetuned')
    bert_pipe = pipeline("text-classification", model=model, tokenizer=tokenizer)
    
    # ML 基线
    with open('models/ml_model.pkl', 'rb') as f:
        ml_model = pickle.load(f)
    with open('models/tfidf_vectorizer.pkl', 'rb') as f:
        tfidf = pickle.load(f)
    
    return bert_pipe, ml_model, tfidf

try:
    bert_pipe, ml_model, tfidf = load_models()
    models_loaded = True
    st.success("✅ 模型加载成功")
except Exception as e:
    models_loaded = False
    st.error(f"❌ 模型加载失败: {e}")

# ── 侧边栏 ──────────────────────────────────────
with st.sidebar:
    st.header("⚙️ 设置")
    model_choice = st.radio(
        "选择模型",
        ["BERT 微调（推荐）", "传统 ML 基线"],
        index=0,
    )
    
    st.divider()
    st.markdown("""
    ### 关于本项目
    - **数据集**: ChnSentiCorp
    - **预训练模型**: bert-base-chinese
    - **技术栈**: Transformers + Streamlit
    """)

# ── 主区域 ──────────────────────────────────────
input_mode = st.radio(
    "输入方式",
    ["单条文本", "批量文本（每行一条）"],
    horizontal=True,
)

if input_mode == "单条文本":
    text = st.text_area(
        "输入文本",
        placeholder="请输入中文文本，例如：这家店的菜很好吃，服务也很周到",
        height=150,
    )
    
    if st.button("🔍 开始分析", type="primary", use_container_width=True):
        if text.strip():
            with st.spinner("分析中..."):
                if model_choice.startswith("BERT"):
                    result = bert_pipe(text)[0]
                    score = result['score']
                    label = result['label']
                    # 判断情感
                    if 'LABEL_1' in label or label == '1':
                        sentiment, emoji, color = "正面", "😊", "#22c55e"
                    else:
                        sentiment, emoji, color = "负面", "😞", "#ef4444"
                else:
                    # ML 模型
                    text_cut = ' '.join(jieba.cut(text))
                    vec = tfidf.transform([text_cut])
                    pred = ml_model.predict(vec)[0]
                    proba = ml_model.predict_proba(vec)[0]
                    if pred == 1:
                        sentiment, emoji, color = "正面", "😊", "#22c55e"
                        score = proba[1]
                    else:
                        sentiment, emoji, color = "负面", "😞", "#ef4444"
                        score = proba[0]
            
            # 显示结果
            st.markdown(f"""
            <div style="
                background: {color}15;
                border: 2px solid {color};
                border-radius: 16px;
                padding: 24px;
                text-align: center;
                margin: 16px 0;
            ">
                <div style="font-size: 48px;">{emoji}</div>
                <div style="font-size: 28px; font-weight: bold; color: {color};">
                    {sentiment}
                </div>
                <div style="font-size: 16px; color: #666; margin-top: 8px;">
                    置信度: {score:.2%}
                </div>
            </div>
            """, unsafe_allow_html=True)
        else:
            st.warning("请输入文本")

else:
    # 批量模式
    batch_text = st.text_area(
        "输入文本（每行一条）",
        placeholder="这家店很好\n服务太差了\n还行吧",
        height=200,
    )
    
    if st.button("🔍 批量分析", type="primary", use_container_width=True):
        texts = [t.strip() for t in batch_text.strip().split('\n') if t.strip()]
        if texts:
            results = []
            for t in texts:
                if model_choice.startswith("BERT"):
                    r = bert_pipe(t)[0]
                    label = r['label']
                    if 'LABEL_1' in label or label == '1':
                        sent = "正面 😊"
                    else:
                        sent = "负面 😞"
                    results.append({
                        "文本": t[:40] + ('...' if len(t) > 40 else ''),
                        "情感": sent,
                        "置信度": f"{r['score']:.1%}",
                    })
                else:
                    t_cut = ' '.join(jieba.cut(t))
                    vec = tfidf.transform([t_cut])
                    pred = ml_model.predict(vec)[0]
                    proba = ml_model.predict_proba(vec)[0]
                    sent = "正面 😊" if pred == 1 else "负面 😞"
                    score = proba[1] if pred == 1 else proba[0]
                    results.append({
                        "文本": t[:40] + ('...' if len(t) > 40 else ''),
                        "情感": sent,
                        "置信度": f"{score:.1%}",
                    })
            
            st.dataframe(results, use_container_width=True)
            
            # 统计
            pos = sum(1 for r in results if "正面" in r['情感'])
            neg = sum(1 for r in results if "负面" in r['情感'])
            st.metric("正面", pos)
            st.metric("负面", neg)

# ── 底部 ────────────────────────────────────────
st.divider()
st.caption("NLP 情感分析课程项目 | 2026")
```

## 8. 常见问题

| 问题 | 解决 |
|------|------|
| `CUDA out of memory` | `per_device_train_batch_size` 改小（8→4→2），`max_length` 改 64 |
| 下载模型太慢 | 设置镜像：`export HF_ENDPOINT=https://hf-mirror.com` |
| 数据集下载失败 | 用本地 CSV 替代方案（见附录） |
| 没装 GPU 版 torch | 训练脚本里 `fp16=False`，慢但能跑 |
| jieba 分词太慢 | 文本不长的话直接用字符级 ngram，去掉 jieba |
